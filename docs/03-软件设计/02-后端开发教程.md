### 4.2 MQTT消息处理

```typescript
// src/utils/mqtt.client.ts
import client from '../config/mqtt';

export class MQTTClient {
  private static instance: MQTTClient;
  private client: any;

  private constructor() {
    this.client = client;
    this.setupEventHandlers();
  }

  public static getInstance(): MQTTClient {
    if (!MQTTClient.instance) {
      MQTTClient.instance = new MQTTClient();
    }
    return MQTTClient.instance;
  }

  private setupEventHandlers() {
    this.client.on('connect', () => {
      console.log('MQTT connected');
      this.subscribeTopics();
    });

    this.client.on('error', (error: Error) => {
      console.error('MQTT error:', error);
    });

    this.client.on('message', (topic: string, message: Buffer) => {
      this.handleMessage(topic, message.toString());
    });
  }

  private subscribeTopics() {
    const topics = [
      'iot/smart/home/status/device/#',
      'iot/smart/home/event/alarm/#',
      'iot/smart/home/event/error/#',
    ];

    topics.forEach(topic => {
      this.client.subscribe(topic, { qos: 1 });
    });
  }

  private handleMessage(topic: string, message: string) {
    console.log(`Received message from ${topic}:`, message);
    
    // 处理消息逻辑
    switch (true) {
      case topic.startsWith('iot/smart/home/status/device/'):
        this.handleDeviceStatus(topic, message);
        break;
      case topic.startsWith('iot/smart/home/event/alarm/'):
        this.handleAlarmEvent(topic, message);
        break;
      case topic.startsWith('iot/smart/home/event/error/'):
        this.handleErrorEvent(topic, message);
        break;
    }
  }

  private handleDeviceStatus(topic: string, message: string) {
    // 处理设备状态更新
  }

  private handleAlarmEvent(topic: string, message: string) {
    // 处理报警事件
  }

  private handleErrorEvent(topic: string, message: string) {
    // 处理错误事件
  }

  public async publish(topic: string, message: string, qos: number = 1): Promise<void> {
    return new Promise((resolve, reject) => {
      this.client.publish(topic, message, { qos }, (err: Error | null) => {
        if (err) {
          reject(err);
        } else {
          resolve();
        }
      });
    });
  }
}
```

## 🔐 认证授权

### 5.1 JWT认证中间件

```typescript
// src/middleware/auth.ts
import { Request, Response, NextFunction } from 'express';
import jwt from 'jsonwebtoken';

export interface AuthRequest extends Request {
  user?: {
    id: number;
    username: string;
    role: string;
  };
}

export const authenticate = (req: AuthRequest, res: Response, next: NextFunction) => {
  const authHeader = req.headers.authorization;
  
  if (!authHeader || !authHeader.startsWith('Bearer ')) {
    return res.status(401).json({ 
      code: 401, 
      message: '未提供认证令牌' 
    });
  }

  const token = authHeader.substring(7);
  
  try {
    const decoded = jwt.verify(token, process.env.JWT_SECRET!);
    req.user = decoded as any;
    next();
  } catch (error) {
    return res.status(401).json({ 
      code: 401, 
      message: '令牌无效或已过期' 
    });
  }
};

export const authorize = (roles: string[]) => {
  return (req: AuthRequest, res: Response, next: NextFunction) => {
    if (!req.user || !roles.includes(req.user.role)) {
      return res.status(403).json({ 
        code: 403, 
        message: '权限不足' 
      });
    }
    next();
  };
};
```

### 5.2 用户登录接口

```typescript
// src/controllers/user.controller.ts
import { Request, Response } from 'express';
import bcrypt from 'bcryptjs';
import jwt from 'jsonwebtoken';
import { UserModel } from '../models/user.model';

export class UserController {
  public async login(req: Request, res: Response) {
    try {
      const { username, password } = req.body;

      const user = await UserModel.findByUsername(username);
      if (!user) {
        return res.status(401).json({ 
          code: 401, 
          message: '用户名或密码错误' 
        });
      }

      const isPasswordValid = await bcrypt.compare(password, user.password);
      if (!isPasswordValid) {
        return res.status(401).json({ 
          code: 401, 
          message: '用户名或密码错误' 
        });
      }

      const token = jwt.sign(
        { id: user.id, username: user.username, role: user.role },
        process.env.JWT_SECRET!,
        { expiresIn: process.env.JWT_EXPIRES_IN }
      );

      res.json({
        code: 200,
        message: '登录成功',
        data: {
          token,
          user: {
            id: user.id,
            username: user.username,
            role: user.role,
          },
        },
      });
    } catch (error) {
      console.error('Login error:', error);
      res.status(500).json({ 
        code: 500, 
        message: '服务器错误' 
      });
    }
  }
}
```

### 6.1 Socket.io配置

```typescript
// src/server.ts
import express from 'express';
import http from 'http';
import { Server } from 'socket.io';
import { deviceController } from './controllers/device.controller';

const app = express();
const server = http.createServer(app);
const io = new Server(server, {
  cors: {
    origin: '*',
    methods: ['GET', 'POST'],
  },
});

// 设备连接处理
io.on('connection', (socket) => {
  console.log('Client connected:', socket.id);

  socket.on('device:register', async (data) => {
    try {
      await deviceController.registerDevice(socket, data);
    } catch (error) {
      socket.emit('error', { message: '设备注册失败' });
    }
  });

  socket.on('device:status', async (data) => {
    try {
      await deviceController.updateDeviceStatus(socket, data);
    } catch (error) {
      socket.emit('error', { message: '状态更新失败' });
    }
  });

  socket.on('disconnect', () => {
    console.log('Client disconnected:', socket.id);
  });
});

server.listen(3000, () => {
  console.log('Server running on port 3000');
});
```

### 7.1 智谱GLM-4集成

```typescript
// src/services/ai.service.ts
import axios from 'axios';

export class AIService {
  private apiKey: string;
  private apiUrl: string;

  constructor() {
    this.apiKey = process.env.AI_API_KEY!;
    this.apiUrl = process.env.AI_API_URL!;
  }

  public async chat(messages: Array<{ role: string; content: string }>) {
    try {
      const response = await axios.post(
        this.apiUrl,
        {
          model: 'glm-4-flash',
          messages: messages,
          temperature: 0.7,
        },
        {
          headers: {
            'Authorization': `Bearer ${this.apiKey}`,
            'Content-Type': 'application/json',
          },
        }
      );

      return response.data;
    } catch (error) {
      console.error('AI chat error:', error);
      throw error;
    }
  }

  public async recognizeIntent(userMessage: string) {
    const prompt = `
    请分析用户的意图，返回JSON格式：
    {
      "intent": "意图名称",
      "entities": {
        "设备": "设备名称",
        "动作": "动作",
        "参数": "参数值"
      }
    }
    
    用户消息：${userMessage}
    `;

    const response = await this.chat([
      { role: 'system', content: '你是一个智能家居控制系统的AI助手，负责理解用户的语音指令' },
      { role: 'user', content: prompt },
    ]);

    return JSON.parse(response.choices[0].message.content);
  }

  public async generateResponse(userMessage: string, context: any) {
    const prompt = `
    你是一个智能家居控制系统的AI助手。
    
    用户消息：${userMessage}
    当前环境信息：${JSON.stringify(context)}
    
    请生成友好的回复。
    `;

    const response = await this.chat([
      { role: 'system', content: '你是一个智能家居控制系统的AI助手，负责与用户进行自然语言对话' },
      { role: 'user', content: prompt },
    ]);

    return response.choices[0].message.content;
  }
}
```
