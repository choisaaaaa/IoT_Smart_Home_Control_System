## 📡 WebSocket集成

### 4.1 WebSocket服务

```typescript
// src/services/websocket.ts
import io from 'socket.io-client'

class WebSocketService {
  private socket: any = null
  private url: string
  private connected: boolean = false

  constructor() {
    this.url = import.meta.env.VITE_WS_URL || 'http://localhost:3000'
  }

  public connect() {
    if (this.connected) return

    this.socket = io(this.url, {
      reconnection: true,
      reconnectionAttempts: 5,
      reconnectionDelay: 1000,
    })

    this.socket.on('connect', () => {
      console.log('WebSocket connected')
      this.connected = true
    })

    this.socket.on('disconnect', () => {
      console.log('WebSocket disconnected')
      this.connected = false
    })

    this.socket.on('error', (error: Error) => {
      console.error('WebSocket error:', error)
    })
  }

  public disconnect() {
    if (this.socket) {
      this.socket.disconnect()
      this.socket = null
      this.connected = false
    }
  }

  public on(event: string, callback: (data: any) => void) {
    this.socket?.on(event, callback)
  }

  public off(event: string, callback?: (data: any) => void) {
    if (callback) {
      this.socket?.off(event, callback)
    } else {
      this.socket?.off(event)
    }
  }

  public emit(event: string, data: any) {
    if (this.socket && this.connected) {
      this.socket.emit(event, data)
    }
  }

  public async emitWithAck(event: string, data: any, timeout: number = 5000) {
    return new Promise((resolve, reject) => {
      if (!this.socket || !this.connected) {
        reject(new Error('WebSocket not connected'))
        return
      }

      const timer = setTimeout(() => {
        reject(new Error('Request timeout'))
      }, timeout)

      this.socket.emit(event, data, (response: any) => {
        clearTimeout(timer)
        resolve(response)
      })
    })
  }
}

export const websocketService = new WebSocketService()
```

### 4.2 实时数据订阅

```typescript
// src/composables/useDeviceStatus.ts
import { ref, onMounted, onUnmounted } from 'vue'
import { websocketService } from '@/services/websocket'

export function useDeviceStatus() {
  const devices = ref<any[]>([])
  const loading = ref(false)

  const fetchDevices = async () => {
    loading.value = true
    try {
      // 从API获取设备列表
      const response = await fetch('/api/devices')
      devices.value = await response.json()
    } catch (error) {
      console.error('Failed to fetch devices:', error)
    } finally {
      loading.value = false
    }
  }

  const updateDeviceStatus = (device: any) => {
    const index = devices.value.findIndex(d => d.id === device.id)
    if (index !== -1) {
      devices.value[index] = { ...devices.value[index], ...device }
    }
  }

  onMounted(() => {
    fetchDevices()

    // 订阅设备状态更新
    websocketService.on('device:status:update', (data: any) => {
      updateDeviceStatus(data)
    })

    // 订阅报警通知
    websocketService.on('alarm:notify', (data: any) => {
      console.log('Alarm:', data)
      // 显示报警提示
    })
  })

  onUnmounted(() => {
    websocketService.off('device:status:update')
    websocketService.off('alarm:notify')
  })

  return {
    devices,
    loading,
    updateDeviceStatus,
  }
}
```

## 📡 API服务

### 5.1 HTTP客户端配置

```typescript
// src/services/api.ts
import axios, { AxiosInstance, AxiosRequestConfig, AxiosResponse } from 'axios'
import { websocketService } from './websocket'

const api: AxiosInstance = axios.create({
  baseURL: import.meta.env.VITE_API_BASE_URL,
  timeout: 10000,
  headers: {
    'Content-Type': 'application/json',
  },
})

// 请求拦截器
api.interceptors.request.use(
  (config: AxiosRequestConfig) => {
    // 添加认证令牌
    const token = localStorage.getItem('token')
    if (token) {
      config.headers.Authorization = `Bearer ${token}`
    }
    return config
  },
  (error) => {
    return Promise.reject(error)
  }
)

// 响应拦截器
api.interceptors.response.use(
  (response: AxiosResponse) => {
    const data = response.data
    if (data.code === 401) {
      // 未授权，跳转登录
      localStorage.removeItem('token')
      window.location.href = '/login'
    }
    return data
  },
  (error) => {
    console.error('API error:', error)
    return Promise.reject(error)
  }
)

// 导出API方法
export const api = {
  // 用户相关
  auth: {
    login: (data: any) => api.post('/auth/login', data),
    logout: () => api.post('/auth/logout'),
    register: (data: any) => api.post('/auth/register', data),
  },

  // 设备相关
  devices: {
    list: (params?: any) => api.get('/devices', { params }),
    get: (id: string) => api.get(`/devices/${id}`),
    create: (data: any) => api.post('/devices', data),
    update: (id: string, data: any) => api.put(`/devices/${id}`, data),
    delete: (id: string) => api.delete(`/devices/${id}`),
    control: (id: string, data: any) => api.post(`/devices/${id}/control`, data),
  },

  // 传感器相关
  sensors: {
    data: (params?: any) => api.get('/sensors/data', { params }),
    history: (params?: any) => api.get('/sensors/history', { params }),
  },

  // AI相关
  ai: {
    chat: (data: any) => api.post('/ai/chat', data),
    recognize: (data: any) => api.post('/ai/recognize', data),
  },
}

export default api
```

```vue
<script setup lang="ts">
import { ref, onMounted, onUnmounted, watch } from 'vue'
import * as echarts from 'echarts'

const props = defineProps<{
  options: any
  width?: string
  height?: string
}>()

const chartRef = ref<HTMLDivElement | null>(null)
let chart: echarts.ECharts | null = null

const initChart = () => {
  if (!chartRef.value) return

  chart = echarts.init(chartRef.value)
  chart.setOption(props.options)
}

const resizeChart = () => {
  chart?.resize()
}

onMounted(() => {
  initChart()
  window.addEventListener('resize', resizeChart)
})

onUnmounted(() => {
  window.removeEventListener('resize', resizeChart)
  chart?.dispose()
})

watch(
  () => props.options,
  (newOptions) => {
    chart?.setOption(newOptions)
  },
  { deep: true }
)
</script>
```

### Q3: 如何处理错误边界？

```vue
<!-- src/components/ErrorBoundary.vue -->
<template>
  <div v-if="hasError">
    <a-result
      status="error"
      title="页面加载失败"
      sub-title="请刷新页面或联系管理员"
    >
      <template #extra>
        <a-button type="primary" @click="reload">
          刷新页面
        </a-button>
      </template>
    </a-result>
  </div>
  <div v-else>
    <slot />
  </div>
</template>

<script setup lang="ts">
import { ref, onErrorCaptured } from 'vue'

const hasError = ref(false)

onErrorCaptured(() => {
  hasError.value = true
  return false
})

const reload = () => {
  window.location.reload()
}
</script>
```
